#include <iostream>
#include <vector>
 
using namespace std;
int main() {
	string s1, s2, s3;
	cin >> s1;
	cin >> s2;
	s3 = s1;
	for (int i = 0; i < s1.length(); i++) {
		
		if (s1[i] == '0' && s2[i] == '0') {
			s3[i] = '0';
		}
		else if (s1[i] == '1' && s2[i] == '1') {
			s3[i] = '0';
		}
		else if (s1[i] == '0' && s2[i] == '1') {
			s3[i] = '1';
		}
		else if (s1[i] == '1' && s2[i] == '0') {
			s3[i] = '1';
		}
	}
	std::cout << s3;
 
}
