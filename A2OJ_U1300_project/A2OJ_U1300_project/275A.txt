#include <iostream>
#include <iomanip>
#include <string.h>
#include <vector>

using namespace std;

int main() {
	int s[3][3];
	bool n[3][3];
	cin >> s[0][0] >> s[0][1] >> s[0][2];
	cin >> s[1][0] >> s[1][1] >> s[1][2];
	cin >> s[2][0] >> s[2][1] >> s[2][2];
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			n[i][j] = 1;
		}
	}
	if (s[0][0] % 2 != 0)
	{
		n[0][0] = !n[0][0];
		n[1][0] = !n[1][0];
		//result[1][1] = !result[1][1];
		n[0][1] = !n[0][1];
	}
	if (s[1][0] % 2 != 0)
	{
		n[0][0] = !n[0][0];
		n[1][0] = !n[1][0];
		n[2][0] = !n[2][0];
		n[1][1] = !n[1][1];

	}
	if (s[2][0] % 2 != 0)
	{

		n[2][0] = !n[2][0];
		n[2][1] = !n[2][1];
		n[1][0] = !n[1][0];
	}
	if (s[0][1] % 2 != 0)
	{
		n[0][0] = !n[0][0];

		n[0][1] = !n[0][1];
		n[1][1] = !n[1][1];

		n[0][2] = !n[0][2];
	}
	if (s[1][1] % 2 != 0)
	{
		n[1][1] = !n[1][1];
		n[0][1] = !n[0][1];
		n[2][1] = !n[2][1];
		n[1][0] = !n[1][0];
		n[1][2] = !n[1][2];
	}
	if (s[2][1] % 2 != 0)
	{

		n[1][1] = !n[1][1];

		n[2][0] = !n[2][0];
		n[2][1] = !n[2][1];
		n[2][2] = !n[2][2];
	}
	if (s[0][2] % 2 != 0)
	{

		n[0][1] = !n[0][1];
		n[0][2] = !n[0][2];
		n[1][2] = !n[1][2];
	}
	if (s[1][2] % 2 != 0)
	{

		n[0][2] = !n[0][2];
		n[1][1] = !n[1][1];
		n[1][2] = !n[1][2];

		n[2][2] = !n[2][2];
	}
	if (s[2][2] % 2 != 0)
	{
		n[2][2] = !n[2][2];
		n[2][1] = !n[2][1];
		n[1][2] = !n[1][2];

	}
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			if (n[i][j]) cout << 1;
			else cout << 0;
		}
		cout << '\n';
	}
	
	
	/*

					if (n[k][l] == 0) {
						n[k][l] == 1;
					}
				}

			}
		}
	}*/
}
