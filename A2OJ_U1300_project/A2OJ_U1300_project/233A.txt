#include <iostream>

int main()
{
	int number;
	cin >> number;
	if (number % 2 == 1)
	{
		cout << -1 << endl;
	}
	else
	{
		cout << "2 1";
		for (int i = 3; i < number; i += 2)
		{
			cout << " " << i + 1 << " " << i;
		}
		cout << endl;
	}
	return 0;
}
